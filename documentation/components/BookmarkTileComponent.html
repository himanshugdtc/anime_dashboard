import { Component, Input, OnInit } from '@angular/core';
import { Bookmark } from 'src/app/shared/bookmark.model';

@Component({
  selector: 'app-bookmark-tile',
  templateUrl: './bookmark-tile.component.html',
  styleUrls: ['./bookmark-tile.component.scss'],
})
export class BookmarkTileComponent implements OnInit {
  @Input() bookmark!: Bookmark;
  tileIconSrc!: string;
  faviconError!: boolean;

  constructor() {}

  ngOnInit(): void {
    this.tileIconSrc = this.bookmark.url.origin + '/favicon.ico';
  }
}

<a [href]="bookmark.url" class="bookmark-tile">
  <div class="tile-icon" *ngIf="!faviconError">
    <img [src]="tileIconSrc" alt="WikiPedia" (error)="faviconError = true" />
  </div>
  <div class="tile-icon fallback" *ngIf="faviconError">
    {{ bookmark.name.toUpperCase()[0] }}
  </div>
  <p class="tile-text">{{ bookmark.name }}</p>
</a>
